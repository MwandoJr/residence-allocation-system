import { Component, OnInit } from '@angular/core';
import { FormGroupDirective, Validators, FormGroup, FormControl } from '@angular/forms';

import { NgForm } from '@angular/forms';
import { AddHouseService } from './shared/add-house.service';
@Component({
  selector: 'app-add-house',
  templateUrl: './add-house.component.html',
  styleUrls: ['./add-house.component.css'],
  providers: [AddHouseService]
})
export class AddHouseComponent implements OnInit {
/**
 * Variables for image preview
 */
  public imagePath;
  imgURL: any;
  public message: string;
  // End of variables

  form = new FormGroup({
    name: new FormControl('', Validators.required),
    location: new FormControl('', Validators.required),
    description: new FormControl('', Validators.required),
    price: new FormControl('', Validators.required),
    ownerscontacts: new FormControl('', Validators.required),
    max: new FormControl('', Validators.required),


  });

  constructor(private addHouseService: AddHouseService) { }

  ngOnInit() {
    this.addHouseService.getData();
    this.resetForm();
  }

  /**
   * Image preview on selecting an image from files
   */
  preview(files) {
    if (files.length === 0){
      return;
    }
    const mimeType = files[0].type;
    if (mimeType.match(/image\/*/) == null) {
      this.message = 'Please select an image.';
      return;
    }
    const reader = new FileReader();
    this.imagePath = files;
    reader.readAsDataURL(files[0]);
    // tslint:disable-next-line: variable-name
    reader.onload = (_event) => {
      this.imgURL = reader.result;
    };
  }
  // end of image preview code

  onSubmit(addHouseForm: NgForm) {
    this.addHouseService.insertRoom(addHouseForm.value);
    this.resetForm(addHouseForm);

  }

  resetForm(addHouseForm?: NgForm) {
    if (addHouseForm != null) {
      addHouseForm.reset();
    }
    this.addHouseService.selectedRoomDetails = {
      $key: null,
      image: '',
      name: '',
      location: '',
      price: '0',
      comment: '',
      contacts: '',
      households: ''
    };
  }

}
