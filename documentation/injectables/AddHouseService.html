import { Injectable } from '@angular/core';

import {AngularFireDatabase, AngularFireList} from 'angularfire2/database'
import {AddHouse} from './add-house.model';
@Injectable({
  providedIn: 'root'
})
/*
*
*
*/
export class AddHouseService {
  //store list of houses
  roomList: AngularFireList<any>;
  //store active houses
  selectedRoomDetails: AddHouse = new AddHouse();
//interact with firebase
  constructor(public firebase :AngularFireDatabase) { }

  //function for get data from db

  getData(){
    this.roomList = this.firebase.list('room_details');
    return this.roomList;

  }

  //function for inserting room details
insertRoom(room : AddHouse){

this.roomList.push({
    name: room.name,
    image: room.image,
    location: room.location,
    price: room.price,
    comment: room.comment,
    contacts: room.contacts,
    households: room.households
})
}

updateRoom(room : AddHouse){
this.roomList.update(room.$key,
{
    name: room.name,
  image: room.image,
    location: room.location,
    price: room.price,
    comment: room.comment,
    contacts: room.contacts,
    households: room.households
})
}
  
}
